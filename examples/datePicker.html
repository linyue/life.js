日期选择器

日期 + 时间 + 限制时间范围

life.datePicker({
    query: '#datetime',
    format: 'yyyy-MM-dd hh:mm:00',
    type: 'datetime',
    startDate: '2014-09-10',
    endDate: '2014-09-25',
    skin: 'blue',
    onClose: function(date, input){
        console.log(date, input)
        alert($("#datetime input").val())
    }
});
